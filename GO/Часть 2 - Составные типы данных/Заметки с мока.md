fmt.Println(arr()) // 10, 1, 2, 3, 4, 5, 6

func arr() []int {

a := []int{0, 1, 2, 3, 4} // len 5 cap 5

a[0] = 1 // data [1, 1, 2, 3, 4]

a = append(a, 5) // data [1, 1, 2, 3, 4, 5, 0, 0, 0, 0] len 6 cap 10

newArr := append(a, 6) // data [1, 1, 2, 3, 4, 5, 6, 0, 0, 0] len 7 cap 10

a[0] = 10 // data [10, 1, 2, 3, 4, 5, 6, 0, 0, 0] len 6 cap 10

return newArr

}
